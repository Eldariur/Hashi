def initialize(orient=:HORIZONTAL,str="UneScalePerso")
              super(orient)
  self.set_name(str)
  @cur_value = 7

  self.set_range 5, 15
  self.set_digits 0
  self.set_size_request 500, 100
  self.value= @cur_value

  self.signal_connect "value-changed" do |w|
      on_changed w
  end
end
