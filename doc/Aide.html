def estCas17?()
  @grille.sommets.each_with_index do |x, i|
    if x.connexionsRestantes() > 0
      compteur = 0
      x.getListeVoisinsNonComplets().each do |v|
        if x.possedeAreteAvec(v)
          if(x.donneAreteAvec(v).estDouble)
            compteur += 0
          else
            if v.connexionsRestantes() > 0
              compteur += 1
            else
              compteur += 0
            end
          end
        else
          if v.connexionsRestantes() > 1
            compteur += 2
          elsif v.connexionsRestantes > 0
            compteur += 1
          else
            compteur += 0
          end
        end
      end
      if compteur == x.connexionsRestantes()
        @position = @grille.getCase(x.position.x, x.position.y)
        return true
      end
    end
  end
  return false
end

def estCas18?()
  @grille.sommets.each_with_index do |x, i|
    if x.connexionsRestantes() == 1
      nb_reliable = 0
      x.getListeVoisinsNonComplets().each do |v|
        reliable = false
        if v.connexionsRestantes() > 1
          nb_reliable += 1
        end
        if v.connexionsRestantes() == 1
          traitement = Array.new()
          traites = Array.new()
          x.getVoisins().each do |c|
            if(c != v)
              traitement.push(c)
            end
          end
          v.getVoisins().each do |c|
            if(c != x)
              traitement.push(c)
            end
          end
          traites.push(x)
          traites.push(v)
          until traitement.empty?() || reliable == true
            traitement.each do |s|
              s.getVoisins().each do |c|
                if(!traites.include?(c) && !traitement.include?(c))
                  traitement.push(c)
                end
              end
              if s.complet
                traitement.delete(s)
                traites.push(s)
              else
                reliable = true
              end
            end
          end
        end
        if reliable
          nb_reliable += 1
        end
      end
      if nb_reliable == 1
        @position = @grille.getCase(x.position.x, x.position.y)
        return true
      end
    end
  end
  return false
end
